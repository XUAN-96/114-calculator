// 設定您的資料夾 ID const FOLDER_ID = "1Qb_NstKnRNF-9yMjma0_QQnuUNidXzXc"; // 1. 讀取功能：回傳資料夾內所有的檔案名稱 function doGet(e) { var folder = DriveApp.getFolderById(FOLDER_ID); var files = folder.getFiles(); var fileList = []; while (files.hasNext()) { var file = files.next(); fileList.push(file.getName()); } return ContentService.createTextOutput(JSON.stringify(fileList)) .setMimeType(ContentService.MimeType.JSON); } // 2. 寫入功能：接收 CSV 並存檔 function doPost(e) { try { // 即使前端 Content-Type 變成 text/plain，這裡一樣能正確解析 JSON var data = JSON.parse(e.postData.contents); var filename = data.filename; var csvContent = data.content; var folder = DriveApp.getFolderById(FOLDER_ID); // 檢查檔案是否已存在，若存在則先刪除舊的 (避免重複) var existingFiles = folder.getFilesByName(filename); while (existingFiles.hasNext()) { existingFiles.next().setTrashed(true); } // 建立新檔案 folder.createFile(filename, csvContent, MimeType.CSV); return ContentService.createTextOutput(JSON.stringify({status: "success"})) .setMimeType(ContentService.MimeType.JSON); } catch (error) { return ContentService.createTextOutput(JSON.stringify({status: "error", msg: error.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
